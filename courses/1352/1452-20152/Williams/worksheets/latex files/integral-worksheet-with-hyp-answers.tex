\question  $\displaystyle \int \frac{2}{x-3}    \, dx$ = $2\ln|x-3| +C$

\question $\displaystyle \int \frac{2}{x-3} + \frac {5}{(x-3)^2}

-\frac{(x-3)^{-2}}{-2}+C$
\vspace{.35in}

\question $\displaystyle \int  \cos^2 x \, \sin x \, dx = -\frac{\cos^3 x}{3}+C$
\vspace{.35in}

\question $\displaystyle \int \frac{5}{\sqrt{x^2 +4}} \, dx = 5\ln\left|
\frac{\sqrt{x^2+4}}{2} + \frac{x}{2}\right| +C$
\vspace{.35in}

\question $\int x e^{2x} \, dx = \frac{1}{2} xe^{2x} -\frac{1}{4}e^{2x}+C$
\vspace{.35in}

\question $\int x e^{x^2} \, dx =\frac{1}{2} e^{x^2}+C$
\vspace{.35in}


\question $\int \sin^2(3x)\, dx =\frac{1}{2}\left(x-\frac{1}{6}\sin(6x)\right)+C$
\vspace{.35in}

\question $\displaystyle \int  \frac{3}{x^2+4x} \, dx = \frac{3}{4}\ln|x|
+\frac{-3}{4}\ln|x+4|+C$
\vspace{.35in}

\question $\displaystyle \int  2x\cosh(3x) \, dx =\frac{2}{3}x\sinh(3x) -
\frac{2}{9}\cosh(3x)+C$
\vspace{.25in}

\question $\displaystyle \int   \sqrt{5-x^2} \, dx=\frac{5}{2}\left(
\sin^{-1}\left(\frac{x}{\sqrt{5}}\right) + \frac{x}{\sqrt{5}}\;\frac{\sqrt{5-x^2}}
{\sqrt{5}}\,\right)+C$
\vspace{.25in}

\question $\displaystyle \int \frac{x}{x^2-25}\, dx=\frac{1}{2}\ln|x^2-25|+C$
\vspace{.35in}

\question $\displaystyle \int e^{2x} \sinh(5x)\, dx =
\frac{25}{21}\left(\frac{1}{5}e^{2x}\cosh(5x)-\frac{2}{25}e^{2x}\sinh(5x)
\right)+C$
\vspace{.25in}

\question $\displaystyle \int   \frac{2x}{x^2+6x+5}\, dx = \int \frac{5}{2(x+5)}-
\frac{1}{2(x+1)}\, dx = \frac{5}{2}\ln|x+5| -\frac{1}{2}\ln|x+1|+C$
\vspace{.35in}

\question $\displaystyle \int  \frac{x-1}{x^2+6x+9} \, dx= 4(x+3)^{-1}+\ln|x+3|+C$
\vspace{.35in}

\question $\displaystyle \int  \frac{1}{x^2+6x +14} \, dx = 
\frac{1}{\sqrt{5}} \tan^{-1}\left(\frac{x+3}{\sqrt{5}}\right)+C$
\vspace{.25in}

\question $\displaystyle \int \frac{1}{\sqrt{x^2+6x+14}}  \, dx =
\ln\left|\frac{{\sqrt{5+(x+3)^2}}}{\sqrt{5}}+\frac{x+3}{\sqrt{5}}\right|+C  $
\vspace{.25in}

\question $\displaystyle \int \frac {2x+3}{x^2+6x +14}   \, dx
=\ln|(x+3)^2+5| -\frac{3}{\sqrt{5}}\tan^{-1}\left(\frac{x+3}{\sqrt{5}}\right)+C$
\vspace{.35in}

\question $\displaystyle \int \frac{3\sin x \, dx}{\cos^2 x + 9} =-\tan^{-1}
\left(\frac{\cos x}{3}\right) +C$
\vspace{.35in}

\question $\displaystyle \int  x^4 \sin(2x) \, dx
= -\frac{1}{2}x^4 \cos(2x) +x^3\sin(2x) +\frac{12}{8}x^2\cos(2x)

\question $\displaystyle \int   \frac{\sin(\ln 3x)}{2x} \, dx

\question $\displaystyle \int   \frac{4x+2}{x^2-x} \, dx

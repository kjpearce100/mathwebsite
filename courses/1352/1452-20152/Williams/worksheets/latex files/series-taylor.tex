\question Find the Maclaurin series for $f(x)= 3x^2 \cos(5x)$.
\vspace{1.5in}
\question Find the Maclaurin series for $\int 3x^2 \cos(5x) \, dx$.
\vspace{.1in} 
\question Find the Maclaurin series for the derivative of $3x^2 \cos(5x)$.
\vspace{.1in}

\question Find the first 4 terms of the Taylor series for $f(x)=x\ln x$ centered
about $a=2$.
\vspace{3.9in}

\question If $\displaystyle
f(x)=\sum_{k=0}^\infty \frac{5k}{k!} x^k$, what is $f^{\prime\prime}(0)$?
\vspace{.2in}

\question Indicate if the following are always true or may be false.
\begin{parts}
\part\_\_\_\_\_\_\_ If $\sum_{k=0}^\infty a_k x^k$ converges when $x=4$,
then it must converge when $x=3$.
\part\_\_\_\_\_\_\_ If $\lim_{k\to\infty} a_k=0$, then $\sum_{k=0}^\infty
a_k$ converges.
\end{parts}

\vspace{.6in}
\large{

\textbf{Definition}.  A sequence $\{a_n\}$ is \textbf{increasing} if $a_{n+1} \geq a_n$ for all $n$.
\vspace{.4in}

\textbf{Definition}.  A sequence $\{a_n\}$ is \textbf{strictly increasing} if $a_{n+1} > a_n$ for all $n$.
\vspace{.4in}

\textbf{Definition}.  A sequence $\{a_n\}$ is \textbf{decreasing} if $a_{n+1} \leq a_n$ for all $n$.
\vspace{.4in}

\textbf{Definition}.  A sequence $\{a_n\}$ is \textbf{strictly decreasing} if $a_{n+1} < a_n$ for all $n$.
\vspace{.4in}

\textbf{Definition}.  A sequence $\{a_n\}$ is \textbf{monotone} if it is either increasing or decreasing.
\vspace{.4in}

\textbf{Definition}.  A sequence $\{a_n\}$ is \textbf{bounded above} by $M$ if $a_n\leq M$ for all $n$.
\vspace{.4in}

\textbf{Definition}.  A sequence $\{a_n\}$ is \textbf{bounded below} by $m$ if $a_n\geq m$ for all $n$.
\vspace{.4in}

\textbf{Definition}.  A sequence $\{a_n\}$ is \textbf{bounded} if it is \emph{both} bounded above and bounded below.
\vspace{.6in}

\textbf{Theorem}.  If a sequence is increasing and bounded above, then it must converge.
\vspace{.4in}

\textbf{Theorem}.  If a sequence is decreasing and bounded below, then it must converge.
\vspace{.4in}

\textbf{Bounded Monotone Convergence Theorem}. A monotone sequence converges if and \hspace*{.5in} only if it is bounded.
\vspace{.4in}

\textbf{Theorem}. If $r$ is a fixed number such that $|r|<1$, then
$\displaystyle \lim_{n \to \infty} r^n =0$. \vspace{.4in}
}
